enabled=0

# ------------------------------------------------------------
# 免費方案（擇一；都要支援「睇相」嘅 vision 模型）
# ------------------------------------------------------------
#
# 1) OpenRouter 免費路由（最容易，建議）
#    開戶拎 key：https://openrouter.ai/
#    enabled=1
#    api_key=sk-or-v1-你的key
#    base_url=https://openrouter.ai/api/v1
#    model=openrouter/free
#    （或指定免費 vision，例如 google/gemini-2.0-flash-exp:free）
#
# 2) Google Gemini 免費額度（Google AI Studio）
#    拎 key：https://aistudio.google.com/apikey
#    enabled=1
#    api_key=你的_Gemini_API_Key
#    base_url=https://generativelanguage.googleapis.com/v1beta/openai
#    model=gemini-2.0-flash
#    json_mode=0
#
# 3) Groq 免費額度（有 vision 模型先用）
#    拎 key：https://console.groq.com/
#    enabled=1
#    api_key=gsk_你的key
#    base_url=https://api.groq.com/openai/v1
#    model=meta-llama/llama-4-scout-17b-16e-instruct
#
# 4) 本機完全免費：Ollama（要自己裝，唔使雲端 key）
#    安裝：https://ollama.com/ 然後執行：ollama pull llava
#    enabled=1
#    api_key=ollama
#    base_url=http://127.0.0.1:11434/v1
#    model=llava
#    json_mode=0
#
# mode 說明：
#   fallback = OCR 讀唔到色號/色名時先叫 AI（省額度，建議）
#   always   = 對色相都問 AI
#   off      = 關閉
#
# 注意：免費雲端通常有每分鐘次數限制；色名仍會優先對照 reference 色號表。
#
# 環境變數亦可：

#   PVH_AI_API_KEY=...
#   PVH_AI_MODE=fallback
#   PVH_AI_MODEL=openrouter/free
#   PVH_AI_BASE_URL=https://openrouter.ai/api/v1
#   PVH_AI_JSON_MODE=1
