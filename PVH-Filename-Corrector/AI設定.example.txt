# AI 色名協助設定
# 複製此檔為 AI設定.txt，擇一方案填 key，再將 enabled=1
# 放喺工具根目錄，或「待改名圖片」資料夾內皆可

# ========== 方案 A：Google Gemini（預設／免費額度）==========
enabled=0
mode=fallback
json_mode=0
api_key=
base_url=https://generativelanguage.googleapis.com/v1beta/openai
model=gemini-2.0-flash

# 拎 key：https://aistudio.google.com/apikey
# 可選模型：gemini-2.0-flash / gemini-2.5-flash / gemini-2.5-pro

# ========== 方案 B：DeepSeek（支援睇相，好平）==========
# 拎 key：https://platform.deepseek.com/
# 用下面整段取代上面（或只改呢幾行）：
#
# enabled=1
# mode=fallback
# json_mode=0
# api_key=sk-你的_DeepSeek_key
# base_url=https://api.deepseek.com
# model=deepseek-v4-flash
#
# 較準可用：model=deepseek-v4-pro
# 環境變數：DEEPSEEK_API_KEY=sk-...

# mode 說明：
#   fallback = OCR 讀唔到色號/色名時先叫 AI（省額度，建議）
#   always   = 對色相都問 AI
#   off      = 關閉
#
# 其他：
# OpenRouter: base_url=https://openrouter.ai/api/v1  model=openrouter/free
